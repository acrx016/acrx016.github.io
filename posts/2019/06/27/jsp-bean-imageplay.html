# 通过bean实现页面图像切换

TIP

进行文件读写相关操作时，最好使用外部tomcat而非编译器内置tomcat

# Bean

public class Play {
	int imageNum = 0; // 当前图片号
	int maxNum; // 最大图片数
	String imageNames[]; // 目录内的所有图片名称
	String imageName; // 展示的图片名称
	String tomcatDir, webDir;

	public Play() {
		File file = new File("");// 取当前目录
		String path = file.getAbsolutePath();// 取当前目录路径
		int index = path.indexOf("bin");// 返回路径中bin首次出现的位置
		tomcatDir = path.substring(0, index);// bin之前的目录路径为tomcat引擎的目录
	}

	public void setWebDir(String s) {
		webDir = s;
		if (imageNames == null) {
			// 获取image目录下的所有图片名称
			File imageDir = new File(tomcatDir + "/webapps/" + webDir + "/image");
			imageNames = imageDir.list();// list方法返回的是文件名称数组
		}
		if (imageNames != null)
			maxNum = imageNames.length;
	}

	public void setImageNum(int n) {
		if (n < 0)
			n = maxNum - 1;
		if (n == maxNum)
			n = 0;
		imageNum = n;
		System.out.print(maxNum);
	}

	public int getImageNum() {
		return imageNum;
	}

	public String getImageName() {
		if (imageNames != null)
			imageName = imageNames[imageNum];
		return imageName;
	}

}

# JSP展示页

<%@ page contentType="text/html; charset=UTF-8"%>
 <jsp:useBean id="img" class="test.bean.Play" scope="request"></jsp:useBean>
<html>
<body>
<%
	String webDir=request.getContextPath();//得到的字符串为/test
	webDir=webDir.substring(1);//从反斜杠的后一个字符开始
 %>
<jsp:setProperty property="webDir" name="img" value="<%=webDir %>"/>
<jsp:setProperty property="imageNum" name="img"	param="imageNum"/>
<img alt="图片" src="image/<%=img.getImageName()%>" width="100" height="100">
<br>
<form action="" method="post">
	<input type="submit" value="pre">
	<input type="hidden" value="<%=img.getImageNum()-1 %>" name="imageNum">
</form>
<form action="" method="post">
	<input type="submit" value="next">
	<input type="hidden" value="<%=img.getImageNum()+1 %>" name="imageNum">
</form>
</body>
</html>

# 简单分析

  • Bean的无参构造目的在于获取image文件夹路径
  • setWebDir方法目的在于获取image文件夹内所有图片的名称数组
  • getImageName方法目的在于获取名称数组中第imageNum张图片的名称
