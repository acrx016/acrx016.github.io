# 知识点

servlet（BeanClass）负责创建request生命周期的bean代码是：

BeanClass bean=new BeanClass();
request.setAttribute("keyword",bean);
1
2

TIP

上述代码会把Bean存放到Tomcat引擎管理的内置对象pageContext中。bean被指定的ID是“keyWord”,scope生命周期是PageContext.Request_SCOPE(request)。

# 代码要求

# View

提供两个JSP页面，一个页面可以输入一元二次方程的系数；另一个页面可以显示一元二次方程的系数和根；

# Model

提供JavaBean存储一元二次方程的系数和根，并提供get/set方法；

# Controler

负责计算一元二次方程的根，将数据存储到JavaBean中，然后重定向到JSP页面负责显示Bean中数据。

# 具体实现

# Model 用于存储一元二次方程数据信息的Bean


//Equation.java

public class Equation {
	double a,b,c;
	String rootOne,rootTwo;
	boolean squareEquation;
	public double getA() {
		return a;
	}
	public void setA(double a) {
		this.a = a;
	}
	public double getB() {
		return b;
	}
	public void setB(double b) {
		this.b = b;
	}
	public double getC() {
		return c;
	}
	public void setC(double c) {
		this.c = c;
	}
	public String getRootOne() {
		return rootOne;
	}
	public void setRootOne(String rootOne) {
		this.rootOne = rootOne;
	}
	public String getRootTwo() {
		return rootTwo;
	}
	public void setRootTwo(String rootTwo) {
		this.rootTwo = rootTwo;
	}
	public boolean isSquareEquation() {
		return squareEquation;
	}
	public void setSquareEquation(boolean squareEquation) {
		this.squareEquation = squareEquation;
	}
}

# Controler 负责计算的Servlet


//HandleCalServlet.java

@WebServlet("/HandleCal")

protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
		Equation equation=new Equation();
		request.setAttribute("equation", equation);
		Double a=Double.parseDouble(request.getParameter("a"));
		Double b=Double.parseDouble(request.getParameter("b"));
		Double c=Double.parseDouble(request.getParameter("c"));
		equation.setA(a);
		equation.setB(b);
		equation.setC(c);
		if(a!=0)
			equation.setSquareEquation(true);
		else
			equation.setSquareEquation(false);
		double disk=b*b-4*a*c;
		if(disk<0) {
			equation.setRootOne("无实数根");
			equation.setRootTwo("无实数根");
		}else {
			double root1=(-b+Math.sqrt(disk)/(2*a));
			double root2=(-b-Math.sqrt(disk)/(2*a));
			equation.setRootOne(""+root1);
			equation.setRootTwo(""+root2);
		}
		RequestDispatcher dispatcher=request.getRequestDispatcher("showparam.jsp");
		dispatcher.forward(request, response);
	}

<jsp:useBean id="equation" class="test.bean.Equation" scope="request"></jsp:useBean>

	二次项系数：<jsp:getProperty property="a" name="equation"/><br>
	一次项系数：<jsp:getProperty property="b" name="equation"/><br>
	常数项：<jsp:getProperty property="c" name="equation"/><br>
	是一元二次方程吗？<jsp:getProperty property="squareEquation" name="equation"/><br>
	方程的两根分别是<jsp:getProperty property="rootOne" name="equation"/><br>
	<jsp:getProperty property="rootTwo" name="equation"/>
